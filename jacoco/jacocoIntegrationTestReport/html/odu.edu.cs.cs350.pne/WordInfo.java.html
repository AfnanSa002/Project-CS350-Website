package odu.edu.cs.cs350.pne;

/**
 * Contains the count, normalized count, and term weight of a word found in a document.
 */
public class WordInfo
{
    /**
     * The number of times a word occurs in a document.
     */
    public int count = 0;

    /**
     * Set to 1 if the word's count is 5 or greater and set to 0 otherwise.
     */
    public int normalizedCount = 0;

    /**
     * The term weight of a word, calculated by multiplying the normalized count
     * by the word's inverse document frequency found in VocabCollection
     */
    public double termWeight = 0.0;
}
