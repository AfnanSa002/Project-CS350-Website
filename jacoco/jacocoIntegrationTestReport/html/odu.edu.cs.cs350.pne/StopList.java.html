package odu.edu.cs.cs350.pne;

import java.util.*;
import java.io.File;
import java.io.FileNotFoundException;
import java.util.Scanner;

/**
 * A StopList contains a list of stopwords (insignificant words).
 */
public class StopList 
{
    /**
     * Contains a list of stopwords (insignificant words).
     */
    public List<String> stopWords = new ArrayList<String>();

    /**
     * Creates a StopList by calling readFile with a default file path.
     */
    public StopList()
    {
        //readFile("src/main/resources/stopwords.txt");
        readFile("acmclass-data/stopwords.txt");
    }

    /**
     * Creates a StopList by calling readFile with a specified file path.
     * @param filePath the file path of a file containing stop words
     */
    public StopList(String filePath)
    {
        readFile(filePath);
    }
    
    /**
     * Reads in a StopList from a file containing stop words
     * @param filePath the file path of a file containing stop words
     */
    public void readFile(String filePath)
    {
        try {
            File stopWordFile = new File(filePath);
            Scanner fileReader = new Scanner(stopWordFile);
            while(fileReader.hasNextLine())
            {
                //Read another line of the stop list
                String data = fileReader.nextLine();
                stopWords.add(data);
            }
            fileReader.close();
        }
        catch (FileNotFoundException e) {
            System.out.println("Error: File was not found.");
            e.printStackTrace();
        }
    }

    /**
     * Returns true if the StopList contains token, false if not.
     * @param token A word that is searched for in StopList
     * @return true if StopList contains token, false if not
     */
    public boolean contains(String token)
    {
        return stopWords.contains(token);
    }
    
    /**
     * Returns a list of stop words
     * @return A list of stop words
     */
    public List<String> getList()
    {
        return stopWords;
    }
    
}
